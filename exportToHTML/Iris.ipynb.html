1    #%% 
2    import keras
3    import pandas as pd
4    from sklearn.model_selection import train_test_split
5    from sklearn.preprocessing import LabelEncoder, StandardScaler
6    import matplotlib.pyplot as plt
7    #%% 
8    url = "https://archive.ics.uci.edu/ml/machine-learning-databases/iris/iris.data"
9    columns = ['sepal_length', 'sepal_width', 'petal_length', 'petal_width', 'species']
10   df = pd.read_csv(url, header=None, names=columns)
11   df = df.dropna()
12   print(df.head())
13   #%% 
14   # Split target and values
15   X = df.iloc[:, 0:4].values
16   y = df['species'].values
17   #%% 
18   # change labels to numerical values
19   encoder = LabelEncoder()
20   y_encoded = encoder.fit_transform(y)
21   
22   # make vector from categories
23   y_categorical = keras.utils.to_categorical(y_encoded)
24   
25   # scale values
26   scaler = StandardScaler()
27   X_scaled = scaler.fit_transform(X)
28   
29   # split data set and ensure that test and train set have same amount of targets and also suffle data.
30   X_train, X_test, y_train, y_test = train_test_split(X_scaled, y_categorical, test_size=0.2, random_state=42,
31                                                       shuffle=True, stratify=y)
32   print("Training data shape:", X_train.shape)
33   print("Test data shape:", X_test.shape)
34   print("Training labels shape:", y_train.shape)
35   print("Test labels shape:", y_test.shape)
36   #%% md 
37   # Building the neural network 
38    
39   #%% 
40   model = keras.Sequential([
41       keras.Input(shape=(4,)),
42       keras.layers.Dense(12, activation='relu'),
43       keras.layers.Dense(3, activation='softmax')
44   ])
45   model.compile(optimizer='adam',
46                 loss='categorical_crossentropy',
47                 metrics=['accuracy'])
48   #%% md 
49   # Train the model 
50   I noticed that i  batch size is raised over 8, models starts to perform poorly 
51   #%% 
52   model_data = model.fit(X_train, y_train, epochs=90, batch_size=8, validation_split=0.1)
53   score = model.evaluate(X_test, y_test, verbose=1)
54   
55   #%% md 
56   # Plot the results 
57    I noticed that 12 neurons and 90 epocs was an optimal to get over 95 % accuracy. If neurons are raised it won affect to results anymore. If epocs are decreased  model starts to perform poorly. 
58   #%% md 
59   | Metric                  | Data used                                                             | What it tells you                                                                                           | 
60   | ----------------------- | --------------------------------------------------------------------- | ----------------------------------------------------------------------------------------------------------- | 
61   | **Training accuracy**   | The data the model was **trained on** (`X_train`, `y_train`)          | How well the model learned patterns **from known examples**                                                 | 
62   | **Validation accuracy** | A held-out portion of the training data (your `validation_split=0.1`) | How well the model performs on **unseen data during training** (used to tune epochs and detect overfitting) | 
63   | **Test accuracy**       | The completely **separate test set** (`X_test`, `y_test`)             | How well the model generalizes to **new, never-before-seen data**                                           | 
64    
65   #%% md 
66   In well trained model: 
67   Training accuracy  ≥  Validation accuracy  ≈  Test accuracy 
68   #%% 
69   
70   print("Test loss:", score[0])

73   print("Final training accuracy:", model_data.history['accuracy'][-1])
